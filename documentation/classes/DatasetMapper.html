import { Dataset } from '@prisma/client';

import { QUERY_CONTEXT_PREFIX, EMBEDDING_MODEL_NAME, EMBEDDING_PROVIDER_NAME } from '@leek/constants';
import { uuid } from '@leek/utils';

import { LeekDataset } from '../../../domain/dataset';
import { generateVectorClassPrefixByDatasetId } from '../../../shared/utils/generate-vector-class-prefix-by-dataset-id.util';

export class DatasetMapper {
  static toDomain(raw: Dataset): LeekDataset {
    const domainEntity = new LeekDataset();

    domainEntity.id = raw.id;
    domainEntity.name = raw.name;
    domainEntity.description = raw.description;
    domainEntity.provider = raw.provider;
    domainEntity.dataSourceType = raw.dataSourceType;
    domainEntity.indexStruct = raw.indexStruct;
    domainEntity.embeddingModel = raw.embeddingModel;
    domainEntity.embeddingModelProvider = raw.embeddingModelProvider;
    domainEntity.retrievalModel = raw.retrievalModel;

    domainEntity.createdAt = raw.createdAt;
    domainEntity.updatedAt = raw.updatedAt;

    return domainEntity;
  }

  static toPersistence(domainEntity: LeekDataset): Dataset {
    const persistenceEntity = {} as Dataset;

    const id = uuid();

    persistenceEntity.id = id;
    persistenceEntity.name = domainEntity.name;
    persistenceEntity.dataSourceType = domainEntity.dataSourceType;
    persistenceEntity.description = domainEntity.description;
    persistenceEntity.indexStruct = domainEntity.indexStruct || {
      type: 'pgvector',
      vectorStore: {
        classPrefix: generateVectorClassPrefixByDatasetId(id),
      },
    };
    persistenceEntity.embeddingModel = domainEntity.embeddingModel || EMBEDDING_MODEL_NAME;
    persistenceEntity.embeddingModelProvider = domainEntity.embeddingModelProvider || EMBEDDING_PROVIDER_NAME;
    persistenceEntity.retrievalModel = domainEntity.retrievalModel || {
      topK: 3,
      scoreThreshold: 0.5,
      scoreThresholdEnabled: false,
    };

    if (!domainEntity.description) {
      persistenceEntity.description = (QUERY_CONTEXT_PREFIX + domainEntity.name)
        .replace('/\n/g', '')
        .replace('/\r/g', '');
    }

    return persistenceEntity;
  }
}
