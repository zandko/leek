import { Prisma, PrismaClient } from '@prisma/client';

import { PaginatedResult } from '@leek/interfaces';

import { LeekDocument } from '../../domain/document';
import { QueryDocumentDto } from '../../dto/query-document.dto';

/**
 * 抽象的 Document 仓库接口。
 * 提供创建、查询、更新、重命名和删除文档的方法。
 */
export abstract class DocumentRepository {
  /**
   * 创建一个新的文档。
   * @param data - 文档数据，不包括 `id`、`createdAt` 和 `updatedAt`。
   * @param prisma - 可选的 Prisma 客户端用于事务。
   * @returns 创建的文档。
   *
   * 动作: create
   * 目标: Document
   * 条件: 无
   */
  abstract createDocument(
    data: Omit<LeekDocument, 'id' | 'createdAt' | 'updatedAt'>,
    prisma?: PrismaClient | Prisma.TransactionClient,
  ): Promise<LeekDocument>;

  /**
   * 根据数据集 ID 查询所有文档。
   * @param datasetId - 数据集的 ID。
   * @returns 一个包含所有相关文档的数组。
   *
   * 动作: findMany
   * 目标: Documents
   * 条件: ByDatasetId
   */
  abstract findManyDocumentsByDatasetId(datasetId: string): Promise<LeekDocument[]>;

  /**
   * 根据数据集 ID 查询所有文档（带分页支持）。
   * @param datasetId - 数据集的 ID。
   * @returns 一个包含所有相关文档的数组。
   *
   * 动作: findMany
   * 目标: Documents
   * 条件: WithPaginationByDatasetId
   */
  abstract findManyDocumentsPaginatedByDatasetId(
    datasetId: string,
    queryDocumentDto: QueryDocumentDto,
  ): Promise<PaginatedResult<LeekDocument>>;

  /**
   * 根据文档 ID 查询单个文档。
   * @param id - 文档的 ID。
   * @returns 找到的文档或 null。
   *
   * 动作: find
   * 目标: Document
   * 条件: ById
   */
  abstract findDocumentById(id: string): Promise<LEEK.Nullable<LeekDocument>>;

  /**
   * 根据文档 ID 更新文档。
   * @param id - 要更新的文档 ID。
   * @param payload - 包含部分更新数据的对象。
   * @param prisma - 可选的 Prisma 客户端用于事务。
   *
   * 动作: update
   * 目标: Document
   * 条件: ById
   */
  abstract updateDocumentById(
    id: string,
    payload: Partial<LeekDocument>,
    prisma?: PrismaClient | Prisma.TransactionClient,
  ): Promise<void>;

  /**
   * 根据文档 ID 删除文档。
   * @param id - 要删除的文档 ID。
   *
   * 动作: delete
   * 目标: Document
   * 条件: ById
   */
  abstract deleteDocumentById(id: string): Promise<void>;

  /**
   * 根据文档 ID 重命名文档。
   * @param id - 文档的 ID。
   * @param name - 新的文档名称。
   * @returns 更新后的文档。
   *
   * 动作: rename
   * 目标: Document
   * 条件: ById
   */
  abstract renameDocumentById(id: string, name: string): Promise<LeekDocument>;

  /**
   * 根据数据集 ID 统计文档数量。
   * @param datasetId - 数据集的 ID。
   * @returns 文档数量。
   *
   * 动作: count
   * 目标: Documents
   * 条件: ByDatasetId
   */
  abstract countDocumentsByDatasetId(datasetId: string): Promise<number>;

  /**
   * 根据名称和数据集 ID 查询单个文档。
   * @param name - 文档的名称。
   * @param datasetId - 数据集的 ID。
   * @returns 找到的文档或 null。
   *
   * 动作: find
   * 目标: Document
   * 条件: ByNameAndDatasetId
   */
  abstract findDocumentByNameAndDatasetId(name: string, datasetId: string): Promise<LEEK.Nullable<LeekDocument>>;

  /**
   * 根据数据集 ID 删除所有相关文档。
   * @param datasetId - 数据集的 ID。
   *
   * 动作: deleteMany
   * 目标: Documents
   * 条件: ByDatasetId
   */
  abstract deleteManyDocumentsByDatasetId(datasetId: string): Promise<void>;
}
