import { ApiProperty, ApiPropertyOptional } from '@nestjs/swagger';

import { Exclude } from 'class-transformer';

export class LeekFile {
  @ApiProperty({
    description: '文件的唯一标识符',
    example: '3f29e2b7-8c9e-4d47-bbfa-a4a7f7b81b9c',
  })
  id: string;

  @ApiPropertyOptional({
    description: '存储类型，例如本地或云存储类型',
    example: 'cos',
    default: 'cos',
  })
  @Exclude()
  storageType?: string = 'cos';

  @ApiProperty({
    description: '文件在存储中的唯一键',
    example: 'upload_files/804f3d62-a2be-4375-8aa6-f0d2d08922ff/989efe03-6d61-4dc2-a9b6-8764eeef8355.pdf',
  })
  @Exclude()
  key: string;

  @ApiProperty({
    description: '文件名称',
    example: '浙江政务服务网 公共支付.pdf',
  })
  name: string;

  @ApiPropertyOptional({
    description: '文件描述',
    example: '这是一个文件描述。',
  })
  description?: string;

  @ApiProperty({
    description: '文件大小（以字节为单位）',
    example: 156819,
  })
  size: number;

  @ApiProperty({
    description: '文件扩展名',
    example: 'pdf',
  })
  extension: string;

  @ApiPropertyOptional({
    description: '文件的 MIME 类型',
    example: 'application/pdf',
  })
  mimeType?: string;

  @ApiPropertyOptional({
    description: '文件是否被使用',
    example: false,
  })
  @Exclude()
  used?: boolean;

  @ApiPropertyOptional({
    description: '文件被使用的时间',
    example: '2023-01-01T00:00:00Z',
  })
  @Exclude()
  usedAt?: Date;

  @ApiPropertyOptional({
    description: '文件的哈希值',
    example: 'c6bd97a16ea91fc05155d82af79516c013c5488f175191372bbb07c7031c77fc',
  })
  @Exclude()
  hash?: string;

  @ApiProperty({
    description: '文件创建时间',
    example: '2023-01-01T00:00:00Z',
  })
  createdAt: Date = new Date();
}
