import { Prisma, PrismaClient } from '@prisma/client';

import { PaginatedResult } from '@leek/interfaces';

import { LeekSegment } from '../../domain/segment';
import { QuerySegmentDto } from '../../dto/query-segment.dto';

/**
 * 抽象的 DocumentSegment 仓库接口。
 * 提供创建、更新、查询和删除文档片段的方法，
 * 支持各种过滤条件和事务选项。
 */
export abstract class DocumentSegmentRepository {
  /**
   * 创建一个新的文档片段。
   * @param data - 包含文档片段数据，但不包括 `id`、`createdAt` 和 `updatedAt`。
   * @param prisma - 可选的 Prisma 客户端用于事务。
   * @returns 创建的文档片段。
   *
   * 动作: create
   * 目标: Segment
   * 条件: 无
   */
  abstract createSegment(
    data: Omit<LeekSegment, 'id' | 'createdAt' | 'updatedAt'>,
    prisma?: PrismaClient | Prisma.TransactionClient,
  ): Promise<LeekSegment>;

  /**
   * 批量创建文档片段并返回结果。
   * @param data - 包含多个文档片段的数据数组，不包括 `id` 和 `createdAt`。
   * @param prisma - 可选的 Prisma 客户端用于事务。
   * @returns 创建的文档片段数组。
   *
   * 动作: createMany
   * 目标: Embeddings
   * 条件: 无
   */
  abstract createManySegmentsAndReturn(data: Omit<LeekSegment, 'id' | 'createdAt'>[]): Promise<LeekSegment[]>;

  /**
   * 批量创建多个文档片段。
   * @param data - 包含多个文档片段数据的数组，不包括 `id`、`createdAt` 和 `updatedAt`。
   * @param prisma - 可选的 Prisma 客户端用于事务。
   *
   * 动作: createMany
   * 目标: Segments
   * 条件: 无
   */
  abstract createManySegments(
    data: Omit<LeekSegment, 'id' | 'createdAt' | 'updatedAt'>[],
    prisma?: PrismaClient | Prisma.TransactionClient,
  ): Promise<void>;

  /**
   * 根据文档 ID 查询所有相关的文档片段。
   * @param documentId - 文档的 ID。
   * @returns 一个包含所有相关文档片段的数组。
   *
   * 动作: findMany
   * 目标: Segments
   * 条件: ByDocumentId
   */
  abstract findManySegmentsByDocumentId(documentId: string): Promise<LeekSegment[]>;

  /**
   * 根据文档集 ID 查询所有段落（带分页支持）。
   * @param datasetId - 文档的 ID。
   * @returns 一个包含所有相关段落的数组。
   *
   * 动作: findMany
   * 目标: Segments
   * 条件: WithPaginationByDocumentId
   */
  abstract findManySegmentsPaginatedByDocumentId(
    documentId: string,
    querySegmentDto: QuerySegmentDto,
  ): Promise<PaginatedResult<LeekSegment>>;

  /**
   * 根据片段 ID 查询单个文档片段。
   * @param id - 文档片段的 ID。
   * @returns 找到的文档片段或 null。
   *
   * 动作: find
   * 目标: Segment
   * 条件: ById
   */
  abstract findSegmentById(id: string): Promise<LEEK.Nullable<LeekSegment>>;

  /**
   * 根据数据集 ID、文档 ID 和哈希值查询单个文档片段。
   * @param datasetId - 数据集的 ID。
   * @param documentId - 文档的 ID。
   * @param indexNodeHash - 节点索引的哈希值。
   * @returns 找到的文档片段或 null。
   *
   * 动作: find
   * 目标: Segment
   * 条件: ByDatasetDocumentAndHash
   */
  abstract findSegmentByDatasetDocumentAndHash(
    datasetId: string,
    documentId: string,
    indexNodeHash: string,
  ): Promise<LEEK.Nullable<LeekSegment>>;

  /**
   * 根据哈希值数组和数据集 ID 查询相关的文档片段。
   * @param hashes - 节点哈希值的数组。
   * @param datasetId - 数据集的 ID。
   * @returns 一个包含所有相关文档片段的数组。
   *
   * 动作: findMany
   * 目标: Segments
   * 条件: ByHashAndDatasetId
   */
  abstract findManySegmentsByHashAndDatasetId(hashes: string[], datasetId: string): Promise<LeekSegment[]>;

  /**
   * 根据数据集 ID 和节点哈希值数组查询相关的哈希值。
   * @param hashes - 节点哈希值的数组。
   * @param datasetId - 数据集的 ID。
   * @returns 一个包含匹配哈希值的数组。
   *
   * 动作: findMany
   * 目标: Hashes
   * 条件: ByDatasetIdAndHashes
   */
  abstract findManyHashesByDatasetIdAndHashes(hashes: string[], datasetId: string): Promise<string[]>;

  /**
   * 根据片段 ID 更新文档片段。
   * @param id - 要更新的文档片段 ID。
   * @param payload - 包含部分更新数据的对象。
   * @param prisma - 可选的 Prisma 客户端用于事务。
   *
   * 动作: update
   * 目标: Segment
   * 条件: ById
   */
  abstract updateSegmentById(
    id: string,
    payload: Partial<LeekSegment>,
    prisma?: PrismaClient | Prisma.TransactionClient,
  ): Promise<void>;

  /**
   * 根据数据集 ID 和节点哈希值数组修改相关的文档片段命中次数。
   * @param hashes - 节点哈希值的数组。
   * @param datasetId - 数据集的 ID。
   * @returns 一个包含匹配哈希值的数组。
   *
   * 动作: updateMany
   * 目标: SegmentsHitCount
   * 条件: ByDatasetIdAndHashes
   */
  abstract updateManyHitCountByDatasetIdAndHashes(hashes: string[], datasetId: string);

  /**
   * 统计某个文档的文档片段数量。
   * @param documentId - 文档的 ID。
   * @returns 文档片段的数量。
   *
   * 动作: count
   * 目标: Segments
   * 条件: ByDocumentId
   */
  abstract countSegmentsByDocumentId(documentId: string): Promise<number>;

  /**
   * 根据片段 ID 删除单个文档片段。
   * @param id - 要删除的文档片段 ID。
   * @param prisma - 可选的 Prisma 客户端用于事务。
   *
   * 动作: delete
   * 目标: Segment
   * 条件: ById
   */
  abstract deleteSegmentById(id: string, prisma?: PrismaClient | Prisma.TransactionClient): Promise<void>;

  /**
   * 根据文档 ID 删除所有相关的文档片段。
   * @param documentId - 文档的 ID。
   *
   * 动作: delete
   * 目标: Segments
   * 条件: ByDocumentId
   */
  abstract deleteManySegmentsByDocumentId(documentId: string): Promise<void>;

  /**
   * 根据数据集 ID 删除所有相关的文档片段。
   * @param datasetId - 数据集的 ID。
   *
   * 动作: delete
   * 目标: Segments
   * 条件: ByDatasetId
   */
  abstract deleteManySegmentsByDatasetId(datasetId: string): Promise<void>;
}
