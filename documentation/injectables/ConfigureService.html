import { Injectable } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';

import { ConfigureAdapter } from './adapter';

@Injectable()
export class ConfigureService implements ConfigureAdapter {
  constructor(private readonly config: ConfigService) {}

  PORT = parseInt(this.config.get('PORT')) || 3000;
  HOST = this.config.get('HOST') || 'localhost';
  TZ = this.config.get('TZ') || 'UTC';
  FALLBACK_LANGUAGE = this.config.get('FALLBACK_LANGUAGE') || 'en-US';

  OPENAI = {
    API_KEY: this.config.get('OPENAI_API_KEY'),
    BASE_URL: this.config.get('OPENAI_API_HOST'),
  };
  TENCENT = {
    SECRET_ID: this.config.get('TENCENT_SECRET_ID'),
    SECRET_KEY: this.config.get('TENCENT_SECRET_KEY'),
    COS_BUCKET: this.config.get('TENCENT_COS_BUCKET'),
    COS_REGION: this.config.get('TENCENT_COS_REGION'),
  };
  SECURITY = {
    RATE_LIMIT_WINDOW_MS: parseInt(this.config.get('RATE_LIMIT_WINDOW_MS'), 10) || 60000,
    RATE_LIMIT_MAX: parseInt(this.config.get('RATE_LIMIT_MAX'), 10) || 100,
  };
}
