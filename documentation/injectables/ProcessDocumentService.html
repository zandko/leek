import { Injectable } from '@nestjs/common';

import { Document } from '@langchain/core/documents';
import pLimit from 'p-limit';

import { ConfigureAdapter } from '@leek/configure';
import { SegmentationEnum, DEFAULT_SEGMENTATION_RULE, DEFAULT_PRE_PROCESSING_RULES, IndexEnum } from '@leek/constants';
import { COSLoader, recursiveChunkSplitter } from '@leek/langchain';
import { createHash, getExtensions, trimLeadingPunctuation } from '@leek/utils';

import { LLMGeneratorService } from './llm-generator.service';
import { ProcessRuleDto } from '../dto/process-rule.dto';

@Injectable()
export class ProcessDocumentService {
  constructor(
    private readonly configureService: ConfigureAdapter,
    private readonly LLMGeneratorService: LLMGeneratorService,
  ) {}

  /**
   * 处理并分段文档
   *
   * 根据提供的规则，从腾讯云 COS 提取文档，对其进行预处理、分段，必要时生成问答对 (QA)。
   *
   * @param {string} key - 文档在腾讯云 COS 中的唯一标识符 (Key)。
   * @param {ProcessRuleDto} processRule - 分段和预处理规则。
   * @param {string} [docForm] - 可选，文档形式（例如 QA 索引类型）。
   * @param {string} [docLanguage] - 可选，文档语言，用于生成 QA 对。
   * @returns {Promise<object>} 包含处理后的文档数组和应用的规则。
   */
  async processDocumentsWithRules(
    key: string,
    processRule: ProcessRuleDto,
    docForm?: string,
    docLanguage?: string,
  ): Promise<{
    processedDocuments: Document[];
    processedRule: ProcessRuleDto;
  }> {
    // 从 COS 提取原始文档
    const docs = await this.extractDocumentsFromCOS(key);

    let { preProcessingRules, segmentation } = processRule;

    if (processRule.mode === SegmentationEnum.Automatic) {
      preProcessingRules = DEFAULT_PRE_PROCESSING_RULES;
      segmentation = DEFAULT_SEGMENTATION_RULE;
    }

    let splitDocs = docs;

    if (docForm === IndexEnum.QA_INDEX && getExtensions(key) === 'xlsx') {
      splitDocs = docs.map((doc) => {
        const [pageContent, answer] = doc.pageContent.split('<<<>>>');
        return new Document({
          pageContent,
          metadata: {
            ...doc.metadata,
            answer,
            hash: createHash(pageContent),
          },
        });
      });
    } else {
      // 根据分段设置拆分文档
      splitDocs = await this.splitDocumentsBySegmentationRules(docs, segmentation);
    }

    splitDocs = await this.splitDocumentsBySegmentationRules(docs, segmentation);

    // 如果文档形式为 QA 索引类型，则生成 QA 对
    if (docForm === IndexEnum.QA_INDEX) {
      splitDocs = await this.generateQaDocumentsForBatch(splitDocs, docLanguage);
    }

    // 应用预处理规则清理分段文档
    const processedDocuments = this.cleanDocumentsByPreprocessingRules(splitDocs, preProcessingRules);

    return {
      processedDocuments,
      processedRule: {
        preProcessingRules,
        segmentation,
        mode: processRule.mode,
      },
    };
  }

  /**
   * 清理文档
   *
   * 根据预处理规则清理文档内容。
   *
   * @param {Document[]} documents - 需要清理的文档数组。
   * @param {ProcessRuleDto['preProcessingRules']} preProcessingRules - 要应用的预处理规则。
   * @returns {Document[]} 清理后的文档数组。
   */
  cleanDocumentsByPreprocessingRules(
    documents: Document[],
    preProcessingRules: ProcessRuleDto['preProcessingRules'],
  ): Document[] {
    const rulesHandlers: Record<string, (text: string) => string> = {
      removeExtraSpaces: (text) => text.replace(/\s+/g, ' ').trim(),
      removeUrlsEmails: (text) => text.replace(/https?:\/\/[^\s]+|www\.[^\s]+|[\w.-]+@[\w.-]+\.[a-z]{2,}/gi, ''),
    };

    // Filter enabled rules and collect the corresponding handler functions
    const handlers = preProcessingRules
      .filter((op) => op.enabled && rulesHandlers[op.id])
      .map((op) => rulesHandlers[op.id]);

    // Filter enabled rules and collect the corresponding handler functions
    return documents.map((document) => {
      const updatedContent = handlers.reduce((content, handler) => handler(content), document.pageContent);
      const pageContent = trimLeadingPunctuation(updatedContent);
      return new Document({
        pageContent,
        metadata: {
          ...document.metadata,
          hash: createHash(pageContent),
        },
      });
    });
  }

  /**
   * 根据分段规则拆分文档
   *
   * @param {Document[]} documents - 需要拆分的文档数组。
   * @param {ProcessRuleDto['segmentation']} segmentation - 分段规则。
   * @returns {Promise<Document[]>} 拆分后的文档数组。
   */
  async splitDocumentsBySegmentationRules(
    documents: Document[],
    segmentation: ProcessRuleDto['segmentation'],
  ): Promise<Document[]> {
    const splitDocs = await recursiveChunkSplitter(documents, {
      chunkSize: segmentation.maxTokens,
      chunkOverlap: segmentation.chunkOverlap,
      separators: segmentation.separator.split(','),
    });
    // Add a hash for each chunk for tracking and uniqueness
    return splitDocs.map(
      (doc) =>
        new Document({
          pageContent: doc.pageContent,
          metadata: {
            ...doc.metadata,
          },
        }),
    );
  }

  /**
   * 从腾讯云 COS 提取文档
   *
   * @param {string} key - 文档在 COS 中的 Key。
   * @returns {Promise<Document[]>} 提取的文档数组。
   */
  async extractDocumentsFromCOS(key: string): Promise<Document[]> {
    const loader = new COSLoader({
      SecretId: this.configureService.TENCENT.SECRET_ID,
      SecretKey: this.configureService.TENCENT.SECRET_KEY,
      Bucket: this.configureService.TENCENT.COS_BUCKET,
      Region: this.configureService.TENCENT.COS_REGION,
      Key: key,
    });
    return loader.load();
  }

  /**
   * 生成文档的问答对 (QA)
   *
   * @param {Document} document - 用于生成 QA 的文档。
   * @param {string} [docLanguage='English'] - QA 对的语言。
   * @returns {Promise<Document[]>} 生成的 QA 文档数组。
   */
  async generateQaDocumentsForSingle(document: Document, docLanguage: string = 'English'): Promise<Document[]> {
    const documents = await this.LLMGeneratorService.generateQaDocumentFromTextAndLanguage(
      document.pageContent,
      docLanguage,
    );
    return documents.map(
      (doc) =>
        new Document({
          pageContent: doc.question,
          metadata: {
            ...document.metadata,
            answer: doc.answer,
            hash: createHash(doc.question),
          },
        }),
    );
  }

  /**
   * 批量生成问答对 (QA)
   *
   * @param {Document[]} documents - 用于生成 QA 的文档数组。
   * @param {string} [docLanguage='English'] - QA 对的语言。
   * @returns {Promise<Document[]>} 生成的 QA 文档数组。
   */
  private async generateQaDocumentsForBatch(
    documents: Document[],
    docLanguage: string = 'English',
  ): Promise<Document[]> {
    const limit = pLimit(10); // Limit concurrency to avoid overloading resources

    const tasks = documents.map((doc) => limit(async () => this.generateQaDocumentsForSingle(doc, docLanguage)));

    const results = await Promise.all(tasks);
    return results.flat();
  }
}
