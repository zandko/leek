import { BadRequestException, Injectable, InternalServerErrorException, Logger } from '@nestjs/common';

import { Transactional } from '@leek/common';
import { EMBEDDING_MODEL_NAME, EMBEDDING_PROVIDER_NAME } from '@leek/constants';
import { PaginatedResult } from '@leek/interfaces';
import { uuid, createHash, encodeAsync } from '@leek/utils';

import { JiebaKeywordService } from './jieba-keyword.service';
import { LeekSegment } from '../domain/segment';
import { CreateDocumentSegmentDto } from '../dto/create-segment.dto';
import { QuerySegmentDto } from '../dto/query-segment.dto';
import { UpdateDocumentSegmentDto } from '../dto/update-segment.dto';
import { DocumentSegmentRepository } from '../infrastructure/persistence/document-segment.repository';
import { DocumentRepository } from '../infrastructure/persistence/document.repository';
import { EmbeddingRepository } from '../infrastructure/persistence/embedding.repository';
import { generateVectorClassPrefixByDatasetId } from '../shared/utils/generate-vector-class-prefix-by-dataset-id.util';

@Injectable()
export class DocumentSegmentService {
  private readonly logger = new Logger(DocumentSegmentService.name);

  constructor(
    private readonly documentRepository: DocumentRepository,
    private readonly documentSegmentRepository: DocumentSegmentRepository,
    private readonly embeddingRepository: EmbeddingRepository,
    private readonly jiebaKeywordService: JiebaKeywordService,
  ) {}

  /**
   * 创建文档分段
   *
   * 根据提供的内容，生成新的文档分段，同时更新文档统计信息和创建嵌入。
   *
   * @param {string} datasetId 数据集 ID
   * @param {string} documentId 文档 ID
   * @param {CreateDocumentSegmentDto} createDocumentSegmentDto 分段创建 DTO
   * @returns {Promise<LeekSegment>} 新创建的文档分段
   */
  @Transactional()
  async createSegment(
    datasetId: string,
    documentId: string,
    createDocumentSegmentDto: CreateDocumentSegmentDto,
  ): Promise<LeekSegment> {
    const { content } = createDocumentSegmentDto;
    const indexNodeHash = createHash(content);

    // 检查是否存在相同哈希的分段
    const existingSegment = await this.documentSegmentRepository.findSegmentByDatasetDocumentAndHash(
      datasetId,
      documentId,
      indexNodeHash,
    );

    if (existingSegment) {
      throw new BadRequestException('已存在相同哈希值的文档分段。');
    }

    try {
      const maxPosition = await this.documentSegmentRepository.countSegmentsByDocumentId(documentId);
      const tokens = await encodeAsync(content);
      const document = await this.documentRepository.findDocumentById(documentId);

      // 更新文档统计信息和生成嵌入
      const createdSegment = await this.documentSegmentRepository.createSegment({
        ...createDocumentSegmentDto,
        datasetId,
        documentId,
        position: maxPosition + 1,
        wordCount: content.length,
        tokens,
        keywords: this.jiebaKeywordService.extractKeywordsFromText(content),
        indexNodeId: uuid(),
        indexNodeHash,
      });

      await this.documentRepository.updateDocumentById(documentId, {
        wordCount: document.wordCount + content.length,
        tokens: document.tokens + tokens,
      });

      await this.embeddingRepository.registerEmbeddings([
        {
          id: uuid(),
          classPrefix: generateVectorClassPrefixByDatasetId(datasetId),
          hash: indexNodeHash,
          modelName: EMBEDDING_MODEL_NAME,
          providerName: EMBEDDING_PROVIDER_NAME,
        },
      ]);

      return createdSegment;
    } catch (error) {
      this.logger.error('创建文档分段失败。', error.stack);
      throw new InternalServerErrorException('创建文档分段时发生错误，请稍后重试。');
    }
  }

  /**
   * 根据 ID 查找文档分段
   *
   * @param {string} segmentId 分段 ID
   * @returns {Promise<LeekSegment | null>} 找到的分段或 null
   */
  async findSegmentById(segmentId: string): Promise<LEEK.Nullable<LeekSegment>> {
    const segment = await this.documentSegmentRepository.findSegmentById(segmentId);

    if (!segment) {
      throw new BadRequestException(`ID 为 ${segmentId} 的文档分段不存在。`);
    }

    return segment;
  }

  /**
   * 查找文档的所有分段
   *
   * @param {string} documentId 文档 ID
   * @returns {Promise<LeekSegment[]>} 文档分段列表
   */
  async findManySegmentsPaginatedByDocumentId(
    documentId: string,
    querySegmentDto?: QuerySegmentDto,
  ): Promise<PaginatedResult<LeekSegment>> {
    return await this.documentSegmentRepository.findManySegmentsPaginatedByDocumentId(documentId, querySegmentDto);
  }

  /**
   * 更新文档分段
   *
   * @param {string} datasetId 数据集 ID
   * @param {string} documentId 文档 ID
   * @param {string} segmentId 分段 ID
   * @param {UpdateDocumentSegmentDto} updateDocumentSegmentDto 分段更新 DTO
   */
  @Transactional()
  async updateSegmentById(
    datasetId: string,
    documentId: string,
    segmentId: string,
    updateDocumentSegmentDto: UpdateDocumentSegmentDto,
  ): Promise<void> {
    const { content } = updateDocumentSegmentDto;
    const indexNodeHash = createHash(content);

    const existingSegment = await this.documentSegmentRepository.findSegmentById(segmentId);

    if (!existingSegment) {
      throw new BadRequestException('文档分段不存在，或内容未发生变化。');
    }

    try {
      const tokens = await encodeAsync(content);
      const document = await this.documentRepository.findDocumentById(documentId);

      // update segment, document, and embeddings
      await this.documentSegmentRepository.updateSegmentById(segmentId, {
        ...updateDocumentSegmentDto,
        tokens,
        wordCount: content.length,
        indexNodeHash,
      });

      await this.documentRepository.updateDocumentById(documentId, {
        wordCount: document.wordCount - existingSegment.wordCount + content.length,
        tokens: document.tokens - existingSegment.tokens + tokens,
      });

      await this.embeddingRepository.deleteManyEmbeddingsByHashAndClassPrefix(
        existingSegment.indexNodeHash,
        generateVectorClassPrefixByDatasetId(datasetId),
      );

      await this.embeddingRepository.registerEmbeddings([
        {
          id: uuid(),
          classPrefix: generateVectorClassPrefixByDatasetId(datasetId),
          hash: indexNodeHash,
          modelName: EMBEDDING_MODEL_NAME,
          providerName: EMBEDDING_PROVIDER_NAME,
        },
      ]);
    } catch (error) {
      this.logger.error('更新文档分段失败。', error.stack);
      throw new InternalServerErrorException('更新文档分段时发生错误，请稍后重试。');
    }
  }

  /**
   * 删除文档分段
   *
   * @param {string} datasetId 数据集 ID
   * @param {string} documentId 文档 ID
   * @param {string} segmentId 分段 ID
   */
  @Transactional()
  async deleteSegmentById(datasetId: string, documentId: string, segmentId: string): Promise<void> {
    const segment = await this.documentSegmentRepository.findSegmentById(segmentId);

    if (!segment) {
      throw new BadRequestException('未找到要删除的文档分段。');
    }
    try {
      const document = await this.documentRepository.findDocumentById(documentId);
      await this.embeddingRepository.deleteManyEmbeddingsByHashAndClassPrefix(
        segment.indexNodeHash,
        generateVectorClassPrefixByDatasetId(datasetId),
      );

      await this.documentSegmentRepository.deleteSegmentById(segmentId);

      await this.documentRepository.updateDocumentById(documentId, {
        wordCount: document.wordCount - segment.wordCount,
        tokens: document.tokens - segment.tokens,
      });
    } catch (error) {
      this.logger.error(`删除 ID 为 ${segmentId} 的文档分段失败。`, error.stack);
      throw new InternalServerErrorException('删除文档分段时发生错误，请稍后重试。');
    }
  }

  /**
   * 禁用文档分段
   *
   * @param {string} segmentId 分段 ID
   */
  async disableSegmentById(segmentId: string) {
    await this.documentSegmentRepository.updateSegmentById(segmentId, {
      enabled: false,
      disabledAt: new Date(),
    });
  }

  /**
   * 启用文档分段
   *
   * @param {string} segmentId 分段 ID
   */
  async enableSegmentById(segmentId: string) {
    await this.documentSegmentRepository.updateSegmentById(segmentId, {
      enabled: true,
    });
  }
}
