import { Injectable } from '@nestjs/common';

import * as nodejieba from 'nodejieba';

import { STOPWORDS } from '@leek/constants';

@Injectable()
export class JiebaKeywordService {
  /**
   * 提取文本中的关键词，并扩展为子标记（Subtokens）。
   * @param {string} text - 要提取关键词的输入文本。
   * @param {number} [maxKeywords=10] - 每段最大提取关键词的数量。
   * @returns {string[]} 包含关键词和扩展子标记的数组。
   */
  extractKeywordsFromText(text: string, maxKeywordsPerChunk: number = 10): string[] {
    const keywords = nodejieba.extract(text, maxKeywordsPerChunk).map((item) => item.word);
    return this.expandKeywordsToSubtokens(new Set<string>(keywords));
  }

  /**
   * 扩展关键词集合，将其分解为子标记，并移除停用词。
   * @private
   * @param {Set<string>} keywords - 要扩展的关键词集合。
   * @returns {string[]} 包含扩展子标记的数组。
   */
  private expandKeywordsToSubtokens(tokens: Set<string>): string[] {
    const results = new Set<string>();

    for (const token of tokens) {
      results.add(token);
      const subTokens = token.match(/\w+/g); // Matches word-like substrings.
      if (subTokens && subTokens.length > 1) {
        subTokens.forEach((subToken) => {
          if (!STOPWORDS.has(subToken)) {
            results.add(subToken);
          }
        });
      }
    }

    return [...results];
  }
}
